1. users/{uid}
   - uid: string
   - email: string
   - displayName: string
   - avatarUrl: string
   - city: string (default: "Guest" for anonymous)

   - createdAt: Timestamp

   - name: string

   - userId: string
   - badgeId: string
   - earnedAt: Timestamp

   - questionText: string
   - imageUrl: string (optional)
   - options: string[] (4 items)
   - correctAnswer: number (0вЂ“3)
   - category: "recyclable"|"non-recyclable"|"hazardous"|"organic"|"e-waste"
   - difficulty: "easy"|"medium"|"hard"
   - explanation: string

6. quizAttempts/{attemptId}
   - userId: string
   - questionId: string
   - selectedAnswer: number
   - isCorrect: boolean
   - pointsEarned: number
   - attemptedAt: Timestamp

7. wasteReports/{reportId}
   - userId: string
   - imageUrl: string (Firebase Storage URL)
   - description: string
   - locationLat: number
   - locationLng: number
   - city: string
   - status: "pending"|"in_review"|"resolved"
   - adminNotes: string
   - createdAt: Timestamp

   - userId: string | null (null = broadcast)
   - title: string
   - body: string
   - type: "alert"|"reward"|"education"|"report_update"
   - isRead: boolean
   - createdAt: Timestamp

FIRESTORE INDEXES (firestore.indexes.json)
в•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђв•ђ

Define composite indexes for:
1. users: city ASC + totalPoints DESC
   (required for leaderboard query)
2. scanRecords: userId ASC + createdAt DESC
3. wasteReports: userId ASC + createdAt DESC
4. wasteReports: status ASC + createdAt DESC (admin filter)
5. notifications: userId ASC + createdAt DESC
6. quizAttempts: userId ASC + attemptedAt DESC

WITHOUT these indexes, Firestore queries will throw errors
and return no data. These MUST be deployed before testing.
Deploy with: firebase deploy --only firestore:indexes

                          && request.auth.uid == uid;
    }

    // Scan records: create if auth; read own only
    match /scanRecords/{id} {
      allow create: if request.auth != null;
      allow read: if request.auth != null
                   && resource.data.userId == request.auth.uid;
    }

    // Badges: read for all auth users; write via Functions only
    match /badges/{id} {
      allow read: if request.auth != null;
      allow write: if false; // Cloud Functions admin SDK only
    }

    // User badges: read own only; write via Functions only
    match /userBadges/{id} {
      allow read: if request.auth != null
                   && resource.data.userId == request.auth.uid;
      allow write: if false;
    }

    // Quiz questions: read for all auth; write via Functions only
    match /quizQuestions/{id} {
      allow read: if request.auth != null;
      allow write: if false;
    }

    // Quiz attempts: create if auth; read own only
    match /quizAttempts/{id} {
      allow create: if request.auth != null;
      allow read: if request.auth != null
                   && resource.data.userId == request.auth.uid;
    }

    // Waste reports: create if auth; read own only
    match /wasteReports/{id} {
      allow create: if request.auth != null;
      allow read: if request.auth != null
                   && resource.data.userId == request.auth.uid;
    }

    // Notifications: read own only
    match /notifications/{id} {
      allow read: if request.auth != null
                   && resource.data.userId == request.auth.uid;
      allow write: if false;
    }
  }
}
